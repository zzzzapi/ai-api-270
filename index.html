Python示例

from openai import OpenAI

client = OpenAI(
    base_url="https://zzzzapi.com/v1",
    api_key="sk-xxxx"
)
stream = client.chat.completions.create(
    model="gpt-4o-mini",

Java示例

import com.theokanning.openai.OpenAiApi;
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import com.theokanning.openai.service.OpenAiService;

import java.util.ArrayList;
import java.util.List;

public class MistralExample {
    public static void main(String[] args) {
        String apiKey = "sk-xxxx";
        String baseUrl = "https://zzzzapi.com/v1";
        
        OpenAiService service = new OpenAiService(apiKey, baseUrl);
        
        List messages = new ArrayList<>();
        messages.add(new ChatMessage("user", "Explain quantum computing in simple terms"));
        
        ChatCompletionRequest request = ChatCompletionRequest.builder()
            .model("mistral-large")
            .messages(messages)
            .build();
        
        service.createChatCompletion(request).getChoices().forEach(choice -> {
            System.out.println(choice.getMessage().getContent());
        });
    }
}
